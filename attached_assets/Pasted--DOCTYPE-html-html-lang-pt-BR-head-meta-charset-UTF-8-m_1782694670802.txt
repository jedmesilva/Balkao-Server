<script>
  // Em produção, este token vem do backend (FastAPI), gerado por usuário/sessão.
  // Aqui está vazio de propósito — substitua pelo connectToken real gerado via API.
  const CONNECT_TOKEN = "COLE_AQUI_O_CONNECT_TOKEN_GERADO_PELO_BACKEND";

  const statusEl = document.getElementById('status');
  const btn = document.getElementById('connectBtn');

  btn.addEventListener('click', () => {
    if (!CONNECT_TOKEN || CONNECT_TOKEN.startsWith("COLE_AQUI")) {
      statusEl.textContent = "Configure um connectToken válido no código para testar.";
      statusEl.className = "status error";
      return;
    }

    btn.disabled = true;
    statusEl.textContent = "Abrindo conexão segura...";
    statusEl.className = "status";

    const pluggyConnect = new PluggyConnect({
      connectToken: CONNECT_TOKEN,
      includeSandbox: true, // essencial para aparecerem os conectores de teste
      onSuccess: (itemData) => {
        statusEl.textContent = "Identidade verificada com sucesso! Pode voltar ao WhatsApp.";
        statusEl.className = "status success";
        console.log("itemId:", itemData.item.id);

        // Em produção: enviar itemData.item.id para o backend do Balkao
        // fetch('/webhook/pluggy-success', { method: 'POST', body: JSON.stringify({ itemId: itemData.item.id }) })
      },
      onError: (error) => {
        statusEl.textContent = "Algo deu errado. Tente novamente pelo WhatsApp.";
        statusEl.className = "status error";
        btn.disabled = false;
        console.error(error);
      },
      onClose: () => {
        btn.disabled = false;
      }
    });

    pluggyConnect.init();
  });
</script>
